fun bubbleSort(arr:Array<Int>, size:Int):Array<Int>{
    var swap = true
    while(swap){
        swap = false
        for(i in 0 until size-1){
            if(arr[i] > arr[i+1]){
                val temp = arr[i]
                arr[i] = arr[i+1]
                arr[i + 1] = temp

                swap = true
            }
        }
    }
    return arr
}

fun main(args: Array<String>) {
    val list = bubbleSort(arrayOf<Int>(2,15,1,8,4),5)
    for (k in list) print("$k ")
}
