fun bubbleSort(arr:Array<Int>, size:Int):Array<Int>{
    var swap: Boolean = true
    while(swap){
        swap = false
        var i: Int;
        for(i in 0 until size-1){
            if(arr[i] > arr[i+1]){
                val temp: Int = arr[i]
                arr[i] = arr[i+1]
                arr[i + 1] = temp

                swap = true
            }
        }
    }
    var i: Int;
     for(i in 0 until 5){
        println(arr[i])
    }
    return arr
}
